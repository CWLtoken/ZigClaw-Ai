const std = @import("std");
const builtin = @import("builtin");

pub const PAGE_SIZE: usize = 4096;

// ==========================================
// 1. IBus 内省总线 (严格物理内存映射)
// 使用 extern struct 保证 C ABI 兼容，配合 @offsetOf 逐字段断言
// 总大小 = 21 有效字节 + 4075 填充 = 4096 (自然对齐为 4，已满足)
// ==========================================
pub const IBusControlPlane = extern struct {
    ssd_heat_version_flip_rate: f32,
    arena_pressure: f32,
    single_codebook_drift: f32,
    latency_context_hash: u32,
    last_latency_ms: u16,
    current_flush_accumulator: u8,
    latency_happened: bool,
    metabolism_valve: u8,
    _pad: [PAGE_SIZE - 21]u8,

    pub fn zero_init() IBusControlPlane {
        return .{
            .ssd_heat_version_flip_rate = 0.0,
            .arena_pressure = 0.0,
            .single_codebook_drift = 0.0,
            .latency_context_hash = 0,
            .last_latency_ms = 0,
            .current_flush_accumulator = 0,
            .latency_happened = false,
            .metabolism_valve = 0,
            ._pad = [_]u8{0} ** (PAGE_SIZE - 21),
        };
    }
};

// ==========================================
// 2. Token 流规范 (网络/跨进程传输)
// 必须是精确 13 字节的紧凑二进制格式
// 禁止 extern struct (C ABI 会填充到 16 字节)
// 禁止 packed struct (大端序、位域语义不确定)
// 唯一安全方案：裸 byte 数组 + 命名视图
// ==========================================
pub const TOKEN_HEADER_SIZE: usize = 13;

pub const TokenStreamHeader = struct {
    const Self = @This();
    raw: [TOKEN_HEADER_SIZE]u8,

    pub fn stream_id(self: *const Self) u64 {
        return std.mem.readInt(u64, &self.raw[0..8], .little);
    }
    pub fn set_stream_id(self: *Self, val: u64) void {
        std.mem.writeInt(u64, &self.raw[0..8], val, .little);
    }
    pub fn total_len(self: *const Self) u32 {
        return std.mem.readInt(u32, &self.raw[8..12], .little);
    }
    pub fn set_total_len(self: *Self, val: u32) void {
        std.mem.writeInt(u32, &self.raw[8..12], val, .little);
    }
    pub fn is_final(self: *const Self) bool {
        return self.raw[12] != 0;
    }
    pub fn set_is_final(self: *Self, val: bool) void {
        self.raw[12] = @intFromBool(val);
    }
    pub fn init() Self {
        return .{ .raw = [_]u8{0} ** TOKEN_HEADER_SIZE };
    }
};

// ==========================================
// 3. 热度数学固化 (动态分段指数衰减)
// ==========================================
pub fn update_heat(current_heat: u16, accessed: bool) u16 {
    var heat = @as(f32, @floatFromInt(current_heat));
    if (accessed) {
        heat += @log(heat + 1.5) * 0.75;
    } else {
        const dyn_decay = 0.00035 + (0.012 / (heat + 2.0));
        heat *= (1.0 - dyn_decay);
    }
    if (heat < 15.0) return 0;
    if (heat > 255.0) heat = 255.0;
    return @as(u16, @intFromFloat(heat));
}

comptime {

    if (@offsetOf(IBusControlPlane, "single_codebook_drift") != 8)
        @compileError("ZC-FATAL: field 2 offset");
    if (@offsetOf(IBusControlPlane, "latency_context_hash") != 12)
        @compileError("ZC-FATAL: field 3 offset");
    if (@offsetOf(IBusControlPlane, "last_latency_ms") != 16)
        @compileError("ZC-FATAL: field 4 offset");
    if (@offsetOf(IBusControlPlane, "current_flush_accumulator") != 18)
        @compileError("ZC-FATAL: field 5 offset");
    if (@offsetOf(IBusControlPlane, "latency_happened") != 19)
